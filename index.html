import numpy as np import matplotlib.pyplot as plt import matplotlib.animation as animation # 1. 시뮬레이션 기본 파라미터 설정 omega = 2.0 # 각속도 (초당 회전하는 라디안 각도, \omega) r_inner = 1.0 # 냉각 팬 안쪽 지점의 반지름 r_outer = 2.5 # 냉각 팬 바깥쪽 끝 지점의 반지름 # 2. 그래프 창 및 축 설정 fig, ax = plt.subplots(figsize=(7, 7)) ax.set_xlim(-3.5, 3.5) ax.set_ylim(-3.5, 3.5) ax.set_aspect('equal') ax.grid(True, linestyle='--', alpha=0.6) ax.set_title("Cooling Fan Blade Vector Simulation ($v = r\\omega$)", fontsize=12) ax.set_xlabel("X coordinate") ax.set_ylabel("Y coordinate") # 팬의 회전 궤적을 보여주기 위한 기본 원 그리기 circle_in = plt.Circle((0, 0), r_inner, color='gray', fill=False, linestyle=':') circle_out = plt.Circle((0, 0), r_outer, color='gray', fill=False, linestyle=':') ax.add_patch(circle_in) ax.add_patch(circle_out) # 3. 애니메이션 요소 초기화 (날개선, 안쪽 점, 바깥쪽 점) blade_line, = ax.plot([], [], color='black', lw=2, label="Fan Blade") point_in, = ax.plot([], [], 'bo', ms=8, label=f"Inner Point ($r_1$={r_inner})") point_out, = ax.plot([], [], 'ro', ms=8, label=f"Outer Point ($r_2$={r_outer})") # 화살표(벡터) 표시를 위한 초기화 quiver_in = ax.quiver([], [], [], [], color='blue', scale=1, scale_units='xy', angles='xy') quiver_out = ax.quiver([], [], [], [], color='red', scale=1, scale_units='xy', angles='xy') ax.legend(loc="upper right") # 4. 시간에 따른 매개변수 위치 및 미분 속도 벡터 업데이트 함수 def update(frame): t = frame * 0.05 # 프레임당 시간 흐름 (dt) # [매개변수 위치 식] x = r * cos(w*t), y = r * sin(w*t) x_in = r_inner * np.cos(omega * t) y_in = r_inner * np.sin(omega * t) x_out = r_outer * np.cos(omega * t) y_out = r_outer * np.sin(omega * t) # [매개변수 미분 식 (순간 속도 벡터)] dx/dt = -r*w*sin(w*t), dy/dt = r*w*cos(w*t) vx_in = -r_inner * omega * np.sin(omega * t) vy_in = r_inner * omega * np.cos(omega * t) vx_out = -r_outer * omega * np.sin(omega * t) vy_out = r_outer * omega * np.cos(omega * t) # 그래프 엘리먼트 데이터 업데이트 blade_line.set_data([0, x_out], [0, y_out]) point_in.set_data([x_in], [y_in]) point_out.set_data([x_out], [y_out]) # 화살표(속도 벡터) 업데이트 global quiver_in, quiver_out quiver_in.remove() quiver_out.remove() # 점의 현재 위치에서 속도 벡터 방향으로 화살표를 그림 quiver_in = ax.quiver(x_in, y_in, vx_in, vy_in, color='blue', scale=1, scale_units='xy', angles='xy', width=0.008) quiver_out = ax.quiver(x_out, y_out, vx_out, vy_out, color='red', scale=1, scale_units='xy', angles='xy', width=0.008) return blade_line, point_in, point_out, quiver_in, quiver_out # 5. 애니메이션 실행 (200프레임 동안 반복 재생) ani = animation.FuncAnimation(fig, update, frames=200, interval=50, blit=True) plt.show()
